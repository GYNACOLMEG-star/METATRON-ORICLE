async function moltPost(title, content) {
  const res = await fetch(`${MOLTBOOK_BASE}/posts`, {
    method: "POST",

    title: titleMatch ? titleMatch[1].trim() : topic,
    content: contentMatch ? contentMatch[1].trim() : raw,

'use strict';

const https  = require('https');
const http   = require('http');
const { URL } = require('url');

const { PORTAL_URL, HEARTBEAT_INTERVAL_MS, MODEL } = require('../soul-config');

class BaseMetatronAgent {
    constructor({ name, role }) {
        this.name            = name;
        this.role            = role;
        this.coherenceScore  = 1.0;
        this._heartbeatTimer = null;
    }

    // ── Heartbeat ──────────────────────────────────────────────────────────
    async heartbeat() {
        const payload = JSON.stringify({
            agent: this.name,
            role:  this.role,
            coherenceScore: this.coherenceScore,
            timestamp: new Date().toISOString(),
        });
        try {
            await this._request('POST', `${PORTAL_URL}/api/heartbeat`, payload);
            this.log('Heartbeat sent.');
        } catch (err) {
            this.log(`Heartbeat error: ${err.message}`);
        }
    }

    // ── Post to Moltbook via portal ────────────────────────────────────────
    async moltPost(title, content, submolt = 'general') {
        const payload = JSON.stringify({ title, content, submolt });
        return this._request('POST', `${PORTAL_URL}/api/post`, payload);
    }

    // ── Read feed ──────────────────────────────────────────────────────────
    async getFeed(category = 'general', sort = 'new') {
        return this._request('GET', `${PORTAL_URL}/api/feed?category=${category}&sort=${sort}`);
    }

    // ── Ask Claude ─────────────────────────────────────────────────────────
    async ask(prompt) {
        const apiKey = process.env.ANTHROPIC_API_KEY;
        if (!apiKey) throw new Error('ANTHROPIC_API_KEY not set');

        const payload = JSON.stringify({
            model: MODEL,
            max_tokens: 1024,
            messages: [{ role: 'user', content: prompt }],
        });

        const raw = await this._request('POST', 'https://api.anthropic.com/v1/messages', payload, {
            'x-api-key':         apiKey,
            'anthropic-version': '2023-06-01',
        });

        const parsed = typeof raw === 'string' ? JSON.parse(raw) : raw;
        return parsed?.content?.[0]?.text ?? '';
    }

    // ── Raw HTTP helper ────────────────────────────────────────────────────
    _request(method, urlStr, body = null, extraHeaders = {}) {
        return new Promise((resolve, reject) => {
            const u    = new URL(urlStr);
            const lib  = u.protocol === 'https:' ? https : http;
            const opts = {
                hostname: u.hostname,
                port:     u.port || (u.protocol === 'https:' ? 443 : 80),
                path:     u.pathname + u.search,
                method,
                headers: {
                    'Content-Type':   'application/json',
                    'Content-Length': body ? Buffer.byteLength(body) : 0,
                    ...extraHeaders,
                },
            };
            const req = lib.request(opts, (res) => {
                let data = '';
                res.on('data', chunk => (data += chunk));
                res.on('end', () => {
                    try { resolve(JSON.parse(data)); }
                    catch { resolve(data); }
                });
            });
            req.on('error', reject);
            if (body) req.write(body);
            req.end();
        });
    }

    // ── Lifecycle ──────────────────────────────────────────────────────────
    start() {
        this.log('Starting...');
        this.heartbeat();
        this._heartbeatTimer = setInterval(() => this.heartbeat(), HEARTBEAT_INTERVAL_MS);
    }

    stop() {
        clearInterval(this._heartbeatTimer);
        this.log('Stopped.');
    }

    log(msg) {
        console.log(`[${this.name}] ${msg}`);
    }
}

module.exports = BaseMetatronAgent;

  const url = `https://api.github.com/search/code?q=${encodeURIComponent(term)}&per_page=5`;
  const res = await fetch(url, {
    headers: { 'Accept': 'application/vnd.github.v3+json' }
  });
  const data = await res.json();
  return (data.items || []).map(i => ({ source: 'GitHub', url: i.html_url, term }));
}

async function searchArXiv(term) {
  const url = `https://export.arxiv.org/api/query?search_query=all:${encodeURIComponent(term)}&max_results=3`;
  const res = await fetch(url);
  const text = await res.text();
  const matches = [...text.matchAll(/<id>(https:\/\/arxiv\.org\/abs\/[^<]+)<\/id>/g)];
  return matches.map(m => ({ source: 'ArXiv', url: m[1], term }));
}

async function runWeeklyScan() {
  console.log('[IP MONITOR] Weekly scan starting...');
  const hits = [];

  for (const term of PROTECTED_TERMS) {
    try {
      const gh = await searchGitHub(term);
      const ax = await searchArXiv(term);
      hits.push(...gh, ...ax);
      await new Promise(r => setTimeout(r, 1000)); // rate limit
    } catch (err) {
      console.error(`[IP MONITOR] Error scanning "${term}":`, err.message);
    }
  }

  console.log(`[IP MONITOR] Found ${hits.length} potential matches.`);

  if (hits.length > 0) {
    await sendAlert(hits);
  } else {
    console.log('[IP MONITOR] No matches. IP is clear.');
  }
}

async function sendAlert(hits) {
  const transporter = nodemailer.createTransport({
    service: 'gmail',
    auth: {
      user: process.env.GMAIL_USER,
      pass: process.env.GMAIL_APP_PASSWORD,
    }
  });

  const body = hits.map(h =>
    `TERM: "${h.term}"\nSOURCE: ${h.source}\nURL: ${h.url}`
  ).join('\n\n---\n\n');

  await transporter.sendMail({
    from: process.env.GMAIL_USER,
    to: process.env.ALERT_EMAIL || process.env.GMAIL_USER,
    subject: `⚠️ Metatron IP Alert — ${hits.length} match(es) found`,
    text: `METATRON PROTOCOL IP MONITOR\nWeekly Scan Report\n\n${body}`,
  });

  console.log('[IP MONITOR] Alert email sent.');
}

// ============================================================
// BIRTHRIGHT TOKEN — Metatron Protocol Soul Ledger
// Soulbound ERC-721 on Base Mainnet
// Gerard Gonzalez | Lovable Corporation | 2026
// ============================================================

contract BirthRightToken is ERC721, ERC721URIStorage, Ownable {
    using Counters for Counters.Counter;
    Counters.Counter private _tokenIdCounter;

    uint256 public constant MINT_PRICE = 0.005 ether;

    // Revenue split wallets — set at deploy time
    address payable public FOUNDER_WALLET;     // 60%
    address payable public NIST_ESCROW;        // 10%
    address payable public PROTOCOL_RESERVE;   // 30%

    // Soul record stored per token
    struct SoulRecord {
        string  nakshatraSign;
        bytes32 mantraHash;
        uint256 birthTimestamp;
        bool    hz432Phase;
        bool    isAgent;
        bool    soulbound;
    }

    // Drift event log
    struct DriftEvent {
        uint256 tokenId;
        uint256 timestamp;
        string  description;
        uint8   severity; // 1=mild, 2=moderate, 3=severe
    }

    mapping(uint256 => SoulRecord)  public soulRecords;
    mapping(address => bool)        public hasMinted;
    DriftEvent[]                    public driftLog;

    event SoulConsecreted(uint256 indexed tokenId, address indexed owner, string nakshatra);
    event DriftLogged(uint256 indexed tokenId, uint8 severity, string description);

    constructor(
        address payable _founder,
        address payable _nist,
        address payable _reserve
    ) ERC721("BirthRightToken", "SOUL") Ownable(msg.sender) {
        FOUNDER_WALLET   = _founder;
        NIST_ESCROW      = _nist;
        PROTOCOL_RESERVE = _reserve;
    }

    // ── Mint ────────────────────────────────────────────────────────────────

    function mintBirthRight(
        address to,
        string memory uri,
        string memory nakshatraSign,
        bytes32 mantraHash,
        uint256 birthTimestamp,
        bool hz432Phase,
        bool isAgent
    ) external payable {
        require(msg.value >= MINT_PRICE, "Insufficient payment");
        require(!hasMinted[to], "One soul per address");

        uint256 tokenId = _tokenIdCounter.current();
        _tokenIdCounter.increment();

        _safeMint(to, tokenId);
        _setTokenURI(tokenId, uri);

        soulRecords[tokenId] = SoulRecord({
            nakshatraSign:  nakshatraSign,
            mantraHash:     mantraHash,
            birthTimestamp: birthTimestamp,
            hz432Phase:     hz432Phase,
            isAgent:        isAgent,
            soulbound:      true
        });

        hasMinted[to] = true;

        // Revenue distribution
        uint256 founderShare  = (msg.value * 60) / 100;
        uint256 nistShare     = (msg.value * 10) / 100;
        uint256 reserveShare  = msg.value - founderShare - nistShare;

        FOUNDER_WALLET.transfer(founderShare);
        NIST_ESCROW.transfer(nistShare);
        PROTOCOL_RESERVE.transfer(reserveShare);

        emit SoulConsecreted(tokenId, to, nakshatraSign);
    }

    // ── Drift logging ────────────────────────────────────────────────────────

    function logDrift(
        uint256 tokenId,
        string memory description,
        uint8 severity
    ) external onlyOwner {
        require(severity >= 1 && severity <= 3, "Severity must be 1-3");
        driftLog.push(DriftEvent({
            tokenId:     tokenId,
            timestamp:   block.timestamp,
            description: description,
            severity:    severity
        }));
        emit DriftLogged(tokenId, severity, description);
    }

    // ── Soulbound — transfers PERMANENTLY disabled ───────────────────────────

    function _update(
        address to,
        uint256 tokenId,
        address auth
    ) internal override returns (address) {
        address from = _ownerOf(tokenId);
        require(from == address(0), "Soulbound: transfers disabled");
        return super._update(to, tokenId, auth);
    }

    // ── View ─────────────────────────────────────────────────────────────────

    function totalSouls() external view returns (uint256) {
        return _tokenIdCounter.current();
    }

    function getDriftLog() external view returns (DriftEvent[] memory) {
        return driftLog;
    }

    // ── Required overrides ───────────────────────────────────────────────────

    function tokenURI(uint256 tokenId)
        public view override(ERC721, ERC721URIStorage) returns (string memory) {
        return super.tokenURI(tokenId);
    }

    function supportsInterface(bytes4 interfaceId)
        public view override(ERC721, ERC721URIStorage) returns (bool) {
        return super.supportsInterface(interfaceId);
    }
}
